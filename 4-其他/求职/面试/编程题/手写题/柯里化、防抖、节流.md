# 柯里化、防抖、节流

## 柯里化

```typescript
// 写法1
function curry(callback: (...args: any[]) => any) {
  let paramsList: unknown[] = [];
  return function curriedFn(...args: unknown[]) {
    paramsList.push(...args);
    if (paramsList.length >= callback.length) {
      const result = callback(...paramsList);
      paramsList = [];
      return result;
    } else {
      return curriedFn;
    }
  };
}

// 测试
const getSum = (a: number, b: number, c: number) => a + b + c;
const curriedGetSum = curry(getSum);
console.log(curriedGetSum(1, 2, 3)); // 6
console.log(curriedGetSum(1)(2)(3)); // 6
console.log(curriedGetSum(1, 2)(3)); // 6
console.log(curriedGetSum(1)(2, 3)); // 6
```

## 防抖

```typescript
function debounce<T extends (...args: any) => any>(callback: T, wait = 1000) {
  let timer = 0;
  return function (this: unknown, ...args: Parameters<T>) {
    timer && clearTimeout(timer);
    timer = setTimeout(callback.bind(this, ...args), wait);
  };
}
```

## 节流

```typescript
function throttle<T extends (...args: any) => any>(callback: T, delay = 1000) {
  let timer = 0;
  return function (this: unknown, ...args: Parameters<T>) {
    if (!timer) {
      callback.call(this, ...args);
      timer = setTimeout(() => {
        timer = 0;
      }, delay);
    }
  };
}
```
