# 深拷贝

## 简单版

```typescript
function deepClone<T>(data: T): T {
  if (typeof data !== 'object' || data === null) {
    return data;
  }

  const clone = (Array.isArray(data) ? [] : {}) as T;
  for (const key in data) {
    Object.hasOwn(data, key) && (clone[key] = deepClone(data[key]));
  }
  return clone;
}
```

## 进阶版

### 解决问题
- 正则表达式对象复制不正确
- 日期对象复制不正确
- 对象嵌套问题

### 代码
```typescript
function deepClone<T>(data: T, hash = new WeakMap()): T {
  if (typeof data !== 'object') return data;
  if (data === null) return null as T;
  if (data instanceof Date) return new Date(data) as T;
  if (data instanceof RegExp) return new RegExp(data) as T;
  if (hash.has(data)) return hash.get(data);

  const clone = (Array.isArray(data) ? [] : {}) as T;
  hash.set(data, clone);
  for (const key in data) {
    Object.hasOwn(data, key) && (clone[key] = deepClone(data[key], hash));
  }
  return clone;
}
```
