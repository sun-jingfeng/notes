```typescript
function searchInsert(nums: number[], target: number): number {
  let left = 0;
  let right = nums.length - 1;
  let result = nums.length;
  while (left <= right) {
    const middle = ((right - left) >> 1) + left;
    if (target === nums[middle]) {
      result = middle;
      break;
    } else if (target < nums[middle]) {
      result = middle;
      right = middle - 1;
    } else {
      left = middle + 1;
    }
  }
  return result;
}
```
