## 一、二叉树的中序遍历

**链接**：[94. 二叉树的中序遍历](https://leetcode.cn/problems/binary-tree-inorder-traversal/)

### 描述

给定二叉树的根节点 `root`，返回它的中序遍历结果。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function inorderTraversal(root: TreeNode | null): number[] {
  const result: number[] = []
  const recursion = (treeNode: TreeNode) => {
    if (treeNode.left) {
      recursion(treeNode.left)
    }
    result.push(treeNode.val)
    if (treeNode.right) {
      recursion(treeNode.right)
    }
  }
  root && recursion(root)
  return result
}
```

## 二、二叉树的最大深度

**链接**：[104. 二叉树的最大深度](https://leetcode.cn/problems/maximum-depth-of-binary-tree/)

### 描述

给定二叉树的根节点 `root`，返回其最大深度。最大深度是指从根节点到最远叶子节点的最长路径上的节点数。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function maxDepth(root: TreeNode | null): number {
  if (root) {
    return Math.max(maxDepth(root.left), maxDepth(root.right)) + 1
  } else {
    return 0
  }
}
```

## 三、翻转二叉树

**链接**：[226. 翻转二叉树](https://leetcode.cn/problems/invert-binary-tree/)

### 描述

给定二叉树的根节点 `root`，翻转这棵二叉树，并返回其根节点。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function invertTree(root: TreeNode | null): TreeNode | null {
  if (root) {
    ;[root.left, root.right] = [root.right, root.left]
    invertTree(root.left)
    invertTree(root.right)
  }
  return root
}
```

## 四、对称二叉树

**链接**：[101. 对称二叉树](https://leetcode.cn/problems/symmetric-tree/)

### 描述

给定二叉树的根节点 `root`，检查它是否轴对称。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function isSymmetric(root: TreeNode | null): boolean {
  return root ? checkMirror(root.left, root.right) : true
}

function checkMirror(node1: TreeNode | null, node2: TreeNode | null): boolean {
  if (node1 === null && node2 === null) return true
  if (node1 === null || node2 === null) return false
  return (
    node1.val === node2.val &&
    checkMirror(node1.left, node2.right) &&
    checkMirror(node1.right, node2.left)
  )
}
```

## 五、二叉树的直径

**链接**：[543. 二叉树的直径](https://leetcode.cn/problems/diameter-of-binary-tree/)

### 描述

给定二叉树的根节点 `root`，返回该树的直径。二叉树的直径是指树中任意两个节点之间最长路径的长度，这条路径可能不经过根节点。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function diameterOfBinaryTree(root: TreeNode | null): number {
  let maxNode = 0
  recursion(root)

  function recursion(node: TreeNode | null): number {
    if (node === null) {
      return 0
    } else {
      const leftMax = recursion(node.left)
      const rightMax = recursion(node.right)
      maxNode = Math.max(maxNode, leftMax + rightMax + 1)
      return Math.max(leftMax, rightMax) + 1
    }
  }

  return maxNode - 1
}
```

## 六、将有序数组转换为二叉搜索树

**链接**：[108. 将有序数组转换为二叉搜索树](https://leetcode.cn/problems/convert-sorted-array-to-binary-search-tree/)

### 描述

给定一个整数数组 `nums`，其中元素已按升序排列，将其转换为一棵高度平衡的二叉搜索树。高度平衡是指每个节点的左右两个子树的高度差的绝对值不超过 1。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function sortedArrayToBST(nums: number[]): TreeNode | null {
  if (nums.length) {
    const midIndex = nums.length >> 1
    const node = new TreeNode(
      nums[midIndex],
      sortedArrayToBST(nums.slice(0, midIndex)),
      sortedArrayToBST(nums.slice(midIndex + 1))
    )
    return node
  } else {
    return null
  }
}
```

## 七、二叉树的层序遍历

**链接**：[102. 二叉树的层序遍历](https://leetcode.cn/problems/binary-tree-level-order-traversal/)

### 描述

给定二叉树的根节点 `root`，返回其层序遍历结果。即逐层从左到右访问所有节点。

### 答案

```typescript
class TreeNode {
  val: number
  left: TreeNode | null
  right: TreeNode | null
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val
    this.left = left === undefined ? null : left
    this.right = right === undefined ? null : right
  }
}

function levelOrder(root: TreeNode | null): number[][] {
  const result: number[][] = []

  const loop = (treeNode: TreeNode | null, depth: number) => {
    if (treeNode) {
      if (!result[depth]) {
        result[depth] = []
      }
      result[depth].push(treeNode.val) // 拓展：如果想 S 形层级遍历，这里更改为 result[depth][depth % 2 === 0 ? 'push' : 'unshift'](treeNode.val)
      loop(treeNode.left, depth + 1)
      loop(treeNode.right, depth + 1)
    }
  }
  loop(root, 0)

  return result
}
```
