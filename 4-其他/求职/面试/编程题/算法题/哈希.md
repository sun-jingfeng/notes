## 两数之和

### 答案

```typescript
function twoSum(nums: number[], target: number): [number, number] {
  const map = new Map<number, number>()
  for (let i = 0; i < nums.length; i++) {
    const res = target - nums[i]
    if (map.has(res)) {
      return [map.get(res) as number, i]
    } else {
      map.set(nums[i], i)
    }
  }
  return [-1, -1]
}
```
