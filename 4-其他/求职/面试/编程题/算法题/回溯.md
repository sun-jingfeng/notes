## 一、全排列

**链接**：[46. 全排列](https://leetcode.cn/problems/permutations/)

### 1.1 描述

给定一个不含重复数字的数组 `nums`，返回其所有可能的全排列。可以按任意顺序返回答案。

### 1.2 答案

```typescript
// 写法 1
function permute(nums: number[]): number[][] {
  const resArr: number[][] = []
  const arr: number[] = []
  loop()

  function loop() {
    nums.forEach(num => {
      if (!arr.includes(num)) {
        arr.push(num)
        if (arr.length === nums.length) {
          resArr.push(arr.slice())
        } else {
          loop()
        }
        arr.pop()
      }
    })
  }

  return resArr
}

// 写法 2
function permute(nums: number[]): number[][] {
  const resArr: number[][] = []
  loop()

  function loop(arr: number[] = []) {
    if (arr.length === nums.length) {
      resArr.push(arr)
    } else {
      nums.forEach(num => {
        if (!arr.includes(num)) {
          loop([...arr, num])
        }
      })
    }
  }

  return resArr
}
```

## 二、子集

**链接**：[78. 子集](https://leetcode.cn/problems/subsets/)

### 2.1 描述

给定一个整数数组 `nums`，数组中的元素互不相同。返回该数组所有可能的子集（幂集）。解集不能包含重复的子集。

### 2.2 答案

```typescript
function subsets(nums: number[]): number[][] {
  const result: number[][] = []
  const arr: number[] = []
  loop()

  function loop(index = 0) {
    if (index >= nums.length) {
      result.push(arr.slice())
    } else {
      arr.push(nums[index])
      loop(index + 1) // 有这个数的情况
      arr.pop()
      loop(index + 1) // 没有这个数的情况
    }
  }

  return result
}
```

## 三、电话号码的字母组合

**链接**：[17. 电话号码的字母组合](https://leetcode.cn/problems/letter-combinations-of-a-phone-number/)

### 3.1 描述

给定一个仅包含数字 `2-9` 的字符串，返回所有它能表示的字母组合。数字到字母的映射与电话按键相同。

### 3.2 答案

```typescript
const map = {
  '1': '*', // ⽤任意字符占个位置，保存结果的时候再去掉，以保证 charList.length 的值正确
  '2': 'abc',
  '3': 'def',
  '4': 'ghi',
  '5': 'jkl',
  '6': 'mno',
  '7': 'pqrs',
  '8': 'tuv',
  '9': 'wxyz',
}

function letterCombinations(digits: string): string[] {
  const result: string[] = []
  const charList: string[] = []

  function loop() {
    for (const char of map[digits.charAt(charList.length) as keyof typeof map]) {
      charList.push(char)
      if (charList.length === digits.length) {
        result.push(charList.filter((item) => item !== '*').join(''))
      } else {
        loop()
      }
      charList.pop()
    }
  }
  
  digits && loop()

  return result
}
```

## 四、组合总和

**链接**：[39. 组合总和](https://leetcode.cn/problems/combination-sum/)

### 4.1 描述

给定一个无重复元素的整数数组 `candidates` 和一个目标整数 `target`，找出 `candidates` 中可以使数字和为目标数 `target` 的所有不同组合。同一数字可以无限次选取。

### 4.2 答案

```typescript
function combinationSum(candidates: number[], target: number): number[][] {
  const result: number[][] = []
  const arr: number[] = []
  recursion()

  function recursion(index = 0, sum = 0) {
    for (let i = index; i < candidates.length; i++) {
      const num = candidates[i]
      arr.push(num)
      if (sum + num === target) {
        result.push(arr.slice())
      } else if (sum + num < target) {
        recursion(i, sum + num)
      }
      arr.pop()
    }
  }

  return result
}
```
