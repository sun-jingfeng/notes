### 1.2 答案

**递归（会超时）**

```typescript
function uniquePaths(m: number, n: number): number {
  if (m === 0 || n === 0) {
    return 0
  }
  if (m === 1 || n === 1) {
    return 1
  }
  return uniquePaths(m - 1, n) + uniquePaths(m, n - 1)
}
```

**迭代**

```typescript
function uniquePaths(m: number, n: number): number {
  const numList = [...new Array(m)].map(() => new Array(n).fill(1))
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (i > 0 && j > 0) {
        numList[i][j] = numList[i - 1][j] + numList[i][j - 1]
      }
    }
  }
  return numList[m - 1][n - 1]
}
```

```typescript
function minPathSum(grid: number[][]): number {
  const row = grid.length
  const col = grid[0].length
  const sumArr = new Array(row).fill(new Array(col))

  for (let i = 0; i < row; i++) {
    for (let j = 0; j < col; j++) {
      if (i === 0 && j === 0) {
        sumArr[i][j] = grid[i][j]
      } else if (i === 0) {
        sumArr[i][j] = grid[i][j] + sumArr[i][j - 1]
      } else if (j === 0) {
        sumArr[i][j] = grid[i][j] + sumArr[i - 1][j]
      } else {
        sumArr[i][j] = grid[i][j] + Math.min(sumArr[i - 1][j], sumArr[i][j - 1])
      }
    }
  }

  return sumArr[row - 1][col - 1]
}
```
