### 答案

```typescript
function subarraySum(nums: number[], k: number): number {
  let count = 0
  for (let i = 0; i < nums.length; i++) {
    let sum = 0
    for (let j = i; j < nums.length; j++) {
      sum += nums[j]
      if (sum === k) {
        count++ // 不中断循环是为了处理后续可能出现的负值数据的情况
      }
    }
  }
  return count
}
```
