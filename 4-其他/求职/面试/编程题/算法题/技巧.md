### 答案

```typescript
function singleNumber(nums: number[]): number {
  let result = 0
  for (const num of nums) {
    result ^= num
  }
  return result
}
```

```typescript
function majorityElement(nums: number[]): number {
  const map: Map<number, number> = new Map()
  for (const num of nums) {
    map.set(num, (map.get(num) ?? 0) + 1)
    if ((map.get(num) ?? 0) > nums.length >> 1) {
      return num
    }
  }
  return NaN
}
```
