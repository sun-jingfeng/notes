### 答案

```typescript
function maxSubArray(nums: number[]): number {
  let max = -Infinity
  let pre = 0
  nums.forEach(num => {
    const result = pre > 0 ? pre + num : num
    max = Math.max(max, result)
    pre = result
  })
  return max
}
```
