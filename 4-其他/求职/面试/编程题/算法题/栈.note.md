```typescript
type IBracket = '(' | ')' | '[' | ']' | '{' | '}'

function isValid(str: string): boolean {
  const stack: IBracket[] = []
  const charArr = str.split('') as IBracket[]
  charArr.forEach(char => {
    if (stack.at(-1) && isPair([stack.at(-1)!, char])) {
      stack.pop()
    } else {
      stack.push(char)
    }
  })
  return stack.length === 0
}

const isPair = (bracketArr: [IBracket, IBracket]): boolean => {
  const result =
    (bracketArr[0] === '(' && bracketArr[1] === ')') ||
    (bracketArr[0] === '[' && bracketArr[1] === ']') ||
    (bracketArr[0] === '{' && bracketArr[1] === '}')
  return result
}
```

```typescript
function dailyTemperatures(temperatures: number[]): number[] {
  const result: number[] = new Array(temperatures.length).fill(0)
  const stack: number[] = [] // 暂未找到下个更高温度天数的索引，这些索引对应的温度值从栈底到栈顶递减
  
  for (let i = 0; i < temperatures.length; i++) {
    // 用当前温度去"解决"stack中所有比当前温度低的索引，直到栈顶索引对应的温度 >= 当前温度
    while (stack.length && temperatures[i] > temperatures[stack.at(-1) as number]) {
      result[stack.at(-1)!] = i - stack.at(-1)!
      stack.pop()
    }
    stack.push(i)
  }
  return result
}
```
