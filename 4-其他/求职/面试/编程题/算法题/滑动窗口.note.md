### 1.2 答案

```typescript
const lengthOfLongestSubstring = function (s: string) {
  const charSet: Set<string> = new Set()
  let maxLength = 0
  let left = 0
  let right = 0
  
  while (left < s.length) {
    while (right < s.length && !charSet.has(s.charAt(right))) {
      charSet.add(s.charAt(right))
      right++
    }
    maxLength = Math.max(maxLength, right - left)
    charSet.delete(s.charAt(left))
    left++
  }
  
  return maxLength
}
```
