### 1.2 答案

```typescript
function maxProfit(prices: number[]): number {
  let maxIncome = -Infinity
  let minPrePrice = Infinity
  prices.forEach(price => {
    minPrePrice = Math.min(minPrePrice, price)
    maxIncome = Math.max(maxIncome, price - minPrePrice)
  })
  return maxIncome
}
```
