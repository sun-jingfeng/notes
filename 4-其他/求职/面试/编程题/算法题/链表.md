## 一、相交链表

**链接**：[160. 相交链表](https://leetcode.cn/problems/intersection-of-two-linked-lists/)

### 描述

给定两个单链表的头节点 `headA` 和 `headB`，找出并返回两个单链表相交的起始节点。若不存在相交节点，返回 `null`。

### 答案

```typescript
class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val
    this.next = next === undefined ? null : next
  }
}

function getIntersectionNode(headA: ListNode | null, headB: ListNode | null): ListNode | null {
  let pA = headA
  let pB = headB
  while (pA !== pB) {
    pA = pA === null ? headB : pA.next
    pB = pB === null ? headA : pB.next
  }
  return pA
}
```

## 二、反转链表

**链接**：[206. 反转链表](https://leetcode.cn/problems/reverse-linked-list/)

### 描述

给定单链表的头节点 `head`，反转链表，并返回反转后的头节点。

### 答案

```typescript
class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val
    this.next = next === undefined ? null : next
  }
}

function reverseList(head: ListNode | null): ListNode | null {
  let prev = null
  let current = head
  while (current) {
    const next = current.next
    current.next = prev
    prev = current
    current = next
  }
  return prev
}
```

## 三、环形链表

**链接**：[141. 环形链表](https://leetcode.cn/problems/linked-list-cycle/)

### 描述

给定链表的头节点 `head`，判断链表中是否存在环。若存在环，返回 `true`；否则返回 `false`。

### 答案

```typescript
class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val
    this.next = next === undefined ? null : next
  }
}

function hasCycle(head: ListNode | null): boolean {
  let slow = head
  let fast = head?.next ?? null
  while (slow !== fast) {
    if (fast === null) {
      return false
    }
    slow = slow?.next ?? null
    fast = fast.next?.next ?? null
  }
  return slow !== null // 不直接返回true，目的是处理首次head为null的特殊情况
}
```

## 四、回文链表

**链接**：[234. 回文链表](https://leetcode.cn/problems/palindrome-linked-list/)

### 描述

给定单链表的头节点 `head`，判断该链表是否为回文链表。若为回文，返回 `true`；否则返回 `false`。

### 答案

```typescript
class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val
    this.next = next === undefined ? null : next
  }
}

// 不能用栈的思路，因为没法处理链表节点为奇数的情况
function isPalindrome(head: ListNode | null): boolean {
  const valueList: number[] = []
  let node = head
  while (node) {
    valueList.push(node.val)
    node = node.next
  }
  for (let i = 0, j = valueList.length - 1; i < j; i++, j--) {
    if (valueList[i] !== valueList[j]) {
      return false
    }
  }
  return true
}
```

## 五、合并两个有序链表

**链接**：[21. 合并两个有序链表](https://leetcode.cn/problems/merge-two-sorted-lists/)

### 描述

将两个升序链表合并为一个新的升序链表并返回。新链表通过拼接给定的两个链表的所有节点组成。

### 答案

```typescript
class ListNode {
  val: number
  next: ListNode | null
  constructor(val?: number, next?: ListNode | null) {
    this.val = val === undefined ? 0 : val
    this.next = next === undefined ? null : next
  }
}

function mergeTwoLists(list1: ListNode | null, list2: ListNode | null): ListNode | null {
  if (list1 === null) {
    return list2
  } else if (list2 === null) {
    return list1
  } else if (list1.val < list2.val) {
    list1.next = mergeTwoLists(list1.next, list2)
    return list1
  } else {
    list2.next = mergeTwoLists(list2.next, list1)
    return list2
  }
}
```
