## 一、算术运算符

### 1. 基本算术运算符

| 运算符 | 说明     | 示例       | 结果   |
| --- | ------ | -------- | ---- |
| `+` | 加法     | `10 + 3` | `13` |
| `-` | 减法     | `10 - 3` | `7`  |
| `*` | 乘法     | `10 * 3` | `30` |
| `/` | 除法     | `10 / 3` | `3`  |
| `%` | 取余（取模） | `10 % 3` | `1`  |

### 2. 使用示例

```java
public class ArithmeticDemo {
    public static void main(String[] args) {
        int a = 10;
        int b = 3;
        
        System.out.println(a + b);  // 13
        System.out.println(a - b);  // 7
        System.out.println(a * b);  // 30
        System.out.println(a / b);  // 3（整数除法，舍弃小数）
        System.out.println(a % b);  // 1（取余数）
    }
}
```

### 3. 整数除法的特点

```java
// 整数除法：结果只保留整数部分，小数直接舍弃（不是四舍五入）
System.out.println(10 / 3);   // 3（不是3.333...）
System.out.println(10 / 4);   // 2（不是2.5）
System.out.println(-10 / 3);  // -3（向零取整）

// 想要小数结果，必须有浮点数参与
System.out.println(10.0 / 3);  // 3.3333...
System.out.println(10 / 3.0);  // 3.3333...
System.out.println((double)10 / 3);  // 3.3333...

// 除以0的情况
// System.out.println(10 / 0);  // ArithmeticException
System.out.println(10.0 / 0);  // Infinity
System.out.println(0.0 / 0);   // NaN
```

### 4. 取余运算详解

```java
// 基本取余
System.out.println(10 % 3);   // 1
System.out.println(10 % 4);   // 2
System.out.println(10 % 10);  // 0

// 负数取余：结果的符号与被除数相同
System.out.println(10 % 3);   // 1
System.out.println(-10 % 3);  // -1（被除数为负，结果为负）
System.out.println(10 % -3);  // 1（被除数为正，结果为正）
System.out.println(-10 % -3); // -1（被除数为负，结果为负）

// 浮点数取余
System.out.println(10.5 % 3);  // 1.5
```

### 5. 取余运算的应用

```java
// 判断奇偶数
int num = 7;
if (num % 2 == 0) {
    System.out.println("偶数");
} else {
    System.out.println("奇数");
}

// 判断能否被整除
if (num % 3 == 0) {
    System.out.println("能被3整除");
}

// 获取各位数字
int n = 12345;
int ge = n % 10;           // 5（个位）
int shi = n / 10 % 10;     // 4（十位）
int bai = n / 100 % 10;    // 3（百位）
int qian = n / 1000 % 10;  // 2（千位）
int wan = n / 10000;       // 1（万位）

// 循环利用（如数组循环）
int[] arr = {1, 2, 3, 4, 5};
for (int i = 0; i < 10; i++) {
    System.out.println(arr[i % arr.length]);  // 循环输出数组元素
}
```

## 二、字符串拼接

### 1. 加号的两种用途

```java
// 1. 数学运算：两边都是数字
int result = 10 + 20;  // 30

// 2. 字符串拼接：任意一边是字符串
String str = "Hello" + "World";  // "HelloWorld"
String msg = "年龄：" + 18;       // "年龄：18"
```

### 2. 字符串拼接规则

```java
// 规则：从左到右依次运算

System.out.println(1 + 2 + "abc");    // "3abc"
// 解析：1+2=3, 3+"abc"="3abc"

System.out.println("abc" + 1 + 2);    // "abc12"
// 解析："abc"+1="abc1", "abc1"+2="abc12"

System.out.println(1 + "abc" + 2);    // "1abc2"
// 解析：1+"abc"="1abc", "1abc"+2="1abc2"

System.out.println(1 + 2 + "abc" + 3 + 4);  // "3abc34"
// 解析：1+2=3, 3+"abc"="3abc", "3abc"+3="3abc3", "3abc3"+4="3abc34"
```

### 3. 使用括号控制运算顺序

```java
System.out.println("结果：" + (10 + 20));  // "结果：30"
System.out.println("结果：" + 10 + 20);    // "结果：1020"
```

### 4. 字符与字符串的区别

```java
// 字符 + 字符 = 数值（ASCII码相加）
System.out.println('a' + 'b');     // 195（97 + 98）
System.out.println('A' + 0);       // 65

// 字符 + 字符串 = 字符串
System.out.println('a' + "b");     // "ab"
System.out.println("" + 'a' + 'b'); // "ab"
```

### 5. 字符串拼接应用

```java
String name = "张三";
int age = 18;
double score = 95.5;

// 拼接输出
System.out.println("姓名：" + name + "，年龄：" + age + "，成绩：" + score);
// 输出：姓名：张三，年龄：18，成绩：95.5

// 使用变量拼接
String info = "姓名：" + name + "，年龄：" + age;
System.out.println(info);
```

## 三、自增自减运算符

### 1. 基本语法

| 运算符  | 说明  | 示例            |
| ---- | --- | ------------- |
| `++` | 自增1 | `a++` 或 `++a` |
| `--` | 自减1 | `a--` 或 `--a` |

### 2. 前置与后置的区别

```java
int a = 10;

// 单独使用：前置和后置效果相同
a++;  // a = 11
++a;  // a = 12
a--;  // a = 11
--a;  // a = 10

// 参与运算时：有区别！
int a = 10;
int b = a++;  // 先赋值，后自增
// b = 10, a = 11

int c = 10;
int d = ++c;  // 先自增，后赋值
// d = 11, c = 11
```

### 3. 详细示例

```java
public class IncrementDemo {
    public static void main(String[] args) {
        // 后置++：先用后加
        int a = 10;
        int b = a++;
        System.out.println("a = " + a);  // 11
        System.out.println("b = " + b);  // 10
        
        // 前置++：先加后用
        int c = 10;
        int d = ++c;
        System.out.println("c = " + c);  // 11
        System.out.println("d = " + d);  // 11
        
        // 在表达式中使用
        int x = 5;
        int y = x++ + ++x;
        // 解析：x++=5(x变6), ++x=7(x变7), 5+7=12
        System.out.println("y = " + y);  // 12
        System.out.println("x = " + x);  // 7
    }
}
```

### 4. 记忆口诀

    ++在前：先加后用（先自增，再参与运算）
    ++在后：先用后加（先参与运算，再自增）

    --同理

### 5. 建议

```java
// ✅ 推荐：单独使用，避免复杂表达式
int a = 10;
a++;
System.out.println(a);

// ❌ 不推荐：在复杂表达式中使用，容易出错
int b = a++ + ++a - a--;  // 难以理解
```

## 四、类型转换

### 1. 自动类型转换（隐式转换）

**规则：小范围 → 大范围，自动转换**

```java
// 转换方向（数据范围从小到大）：
// byte → short → int → long → float → double
//                 ↑
//               char

// 示例
byte b = 10;
int i = b;      // byte → int，自动转换

int num = 100;
double d = num; // int → double，自动转换

char c = 'A';
int code = c;   // char → int，自动转换（得到 ASCII 码 65）

// 注意：long → float 可以自动转换（虽然 long 是 8 字节，float 是 4 字节）
// 因为 float 的表示范围更大，但可能丢失精度
long l = 123456789012345L;
float f = l;  // 自动转换，但可能丢失精度
```

### 2. 强制类型转换（显式转换）

**规则：大范围 → 小范围，需要强制转换**

**格式：** `目标类型 变量名 = (目标类型) 值;`

```java
double d = 3.14;
int i = (int) d;  // i = 3（直接截断小数）

int big = 200;
byte b = (byte) big;  // 可能溢出

long l = 100L;
int n = (int) l;  // long → int
```

### 3. 精度丢失详解

```java
// 1. 浮点数转整数：直接截断小数部分（不是四舍五入）
double d1 = 3.99;
int i1 = (int) d1;  // i1 = 3

double d2 = -3.99;
int i2 = (int) d2;  // i2 = -3（向零取整）

// 2. 大范围整数转小范围：可能溢出
int big = 200;
byte b = (byte) big;  // b = -56（溢出）

// 溢出原理：200 的二进制是 11001000
// byte 只取低 8 位，11001000 作为有符号数是 -56

int big2 = 128;
byte b2 = (byte) big2;  // b2 = -128

int big3 = 256;
byte b3 = (byte) big3;  // b3 = 0

// 3. long → float 可能丢失精度
long l = 123456789012345L;
float f = l;
System.out.println(f);  // 1.23456789E14（精度丢失）

// 4. int → float 也可能丢失精度
int i = 123456789;
float f2 = i;
System.out.println(f2);  // 1.23456792E8（精度丢失）
```

### 4. 运算中的类型转换

```java
// 1. byte/short/char 运算时自动转为 int
byte b1 = 10;
byte b2 = 20;
// byte b3 = b1 + b2;  // ❌ 错误！结果是 int
int b3 = b1 + b2;      // ✅ 正确
byte b4 = (byte)(b1 + b2);  // ✅ 强制转换

// 2. 不同类型运算，转为范围大的类型
int i = 10;
double d = 3.14;
double result = i + d;  // int + double = double

// 3. 常量优化（编译时计算）
byte b = 10 + 20;  // ✅ 正确！编译器知道 10+20=30，在 byte 范围内
byte b5 = 100 + 100;  // ❌ 错误！200 超出 byte 范围

// 4. final 常量也会优化
final int x = 10;
final int y = 20;
byte b6 = x + y;  // ✅ 正确！x 和 y 是 final，编译器可以计算
```

### 5. 类型转换总结表

| 转换类型 | 方向  | 语法      | 风险        |
| ---- | --- | ------- | --------- |
| 自动转换 | 小→大 | 直接赋值    | 无风险       |
| 强制转换 | 大→小 | `(类型)值` | 可能丢失精度或溢出 |

**数据类型范围（从小到大）：**
`byte` < `short` < `int` < `long` < `float` < `double`

```java
int a = 10;  // 将10赋值给变量a
```

### 2. 复合赋值运算符

| 运算符  | 说明    | 示例       | 等价于         |
| ---- | ----- | -------- | ----------- |
| `+=` | 加后赋值  | `a += 5` | `a = a + 5` |

```java
public class AssignmentDemo {
```

## 六、关系运算符（比较运算符）

### 1. 关系运算符列表

| 运算符  | 说明   | 示例       | 结果      |
| ---- | ---- | -------- | ------- |
| `==` | 等于   | `5 == 5` | `true`  |
| `!=` | 不等于  | `5 != 3` | `true`  |
| `>`  | 大于   | `5 > 3`  | `true`  |
| `<`  | 小于   | `5 < 3`  | `false` |
| `>=` | 大于等于 | `5 >= 5` | `true`  |
| `<=` | 小于等于 | `5 <= 3` | `false` |

### 2. 使用示例

```java
public class RelationalDemo {
    public static void main(String[] args) {
        int a = 10;
        int b = 20;
        
        System.out.println(a == b);  // false
        System.out.println(a != b);  // true
        System.out.println(a > b);   // false
        System.out.println(a < b);   // true
        System.out.println(a >= 10); // true
        System.out.println(a <= 10); // true
        
        // 关系运算符的结果是 boolean 类型
        boolean result = a < b;
        System.out.println(result);  // true
    }
}
```

### 3. 浮点数比较问题

```java
// 浮点数精度问题
double a = 0.1 + 0.2;
double b = 0.3;
System.out.println(a);         // 0.30000000000000004
System.out.println(a == b);    // false（不相等！）

// 解决方案1：使用误差范围比较
double epsilon = 1e-10;
if (Math.abs(a - b) < epsilon) {
    System.out.println("相等");
}

// 解决方案2：使用 BigDecimal
import java.math.BigDecimal;
BigDecimal bd1 = new BigDecimal("0.1").add(new BigDecimal("0.2"));
BigDecimal bd2 = new BigDecimal("0.3");
System.out.println(bd1.compareTo(bd2) == 0);  // true
```

### 4. 注意事项

```java
// 1. 区分 = 和 ==
int a = 10;    // 赋值
a == 10;       // 比较

// 2. 关系运算符的结果是 boolean

// 3. 不能连续使用（与数学不同）
int x = 5;
// if (1 < x < 10)  // ❌ 错误！
if (x > 1 && x < 10)  // ✅ 正确

// 4. 引用类型比较
String s1 = new String("hello");
String s2 = new String("hello");
System.out.println(s1 == s2);      // false（比较地址）
System.out.println(s1.equals(s2)); // true（比较内容）
```

## 七、逻辑运算符

### 1. 逻辑运算符列表

| 运算符    | 说明      | 示例                | 结果      |
| ------ | ------- | ----------------- | ------- |
| `&`    | 与（AND）  | `true & false`    | `false` |
| `\|`   | 或（OR）   | `true \| false`   | `true`  |
| `!`    | 非（NOT）  | `!true`           | `false` |
| `^`    | 异或（XOR） | `true ^ false`    | `true`  |
| `&&`   | 短路与     | `true && false`   | `false` |
| `\|\|` | 短路或     | `true \|\| false` | `true`  |

### 2. 逻辑运算真值表

#### 与运算（&、&&）

| A     | B     | A & B |
| ----- | ----- | ----- |
| true  | true  | true  |
| true  | false | false |
| false | true  | false |
| false | false | false |

**口诀：全真才真，有假则假**

#### 或运算（|、||）

| A     | B     | A \| B |
| ----- | ----- | ------ |
| true  | true  | true   |
| true  | false | true   |
| false | true  | true   |
| false | false | false  |

**口诀：有真则真，全假才假**

#### 非运算（!）

| A     | !A    |
| ----- | ----- |
| true  | false |
| false | true  |

**口诀：取反**

#### 异或运算（^）

| A     | B     | A ^ B |
| ----- | ----- | ----- |
| true  | true  | false |
| true  | false | true  |
| false | true  | true  |
| false | false | false |

**口诀：相同为假，不同为真**

### 3. 短路与（&&）和短路或（||）

```java
// 短路与 &&：左边为 false，右边不执行
int a = 10;
int b = 20;
boolean result = (a > b) && (++a > 0);
System.out.println(result);  // false
System.out.println(a);       // 10（++a 没有执行）

// 短路或 ||：左边为 true，右边不执行
boolean result2 = (a < b) || (++a > 0);
System.out.println(result2); // true
System.out.println(a);       // 10（++a 没有执行）

// 普通与 &：两边都执行
boolean result3 = (a > b) & (++a > 0);
System.out.println(a);       // 11（++a 执行了）
```

### 4. 短路运算符对比

| 运算符    | 特点               | 效率 |
| ------ | ---------------- | -- |
| `&`    | 两边都执行            | 低  |
| `&&`   | 左边 false 时，右边不执行 | 高  |
| `\|`   | 两边都执行            | 低  |
| `\|\|` | 左边 true 时，右边不执行  | 高  |

**推荐使用短路运算符 `&&` 和 `||`**

### 5. 使用示例

```java
public class LogicalDemo {
    public static void main(String[] args) {
        int age = 25;
        double salary = 8000;
        
        // 与运算：同时满足
        if (age >= 18 && salary >= 5000) {
            System.out.println("可以申请信用卡");
        }
        
        // 或运算：满足其一
        if (age < 18 || age > 60) {
            System.out.println("不在工作年龄范围");
        }
        
        // 非运算：取反
        boolean isStudent = false;
        if (!isStudent) {
            System.out.println("不是学生");
        }
        
        // 组合使用
        int score = 85;
        if (score >= 60 && score < 90) {
            System.out.println("成绩中等");
        }
        
        // 短路特性的应用：避免空指针
        String str = null;
        if (str != null && str.length() > 0) {
            // 如果 str 为 null，不会执行 str.length()，避免 NullPointerException
        }
    }
}
```

## 八、位运算符

### 1. 位运算符列表

| 运算符   | 说明      | 示例         | 结果    |
| ----- | ------- | ---------- | ----- |
| `&`   | 按位与     | `5 & 3`    | `1`   |
| `\|`  | 按位或     | `5 \| 3`   | `7`   |
| `^`   | 按位异或    | `5 ^ 3`    | `6`   |
| `~`   | 按位取反    | `~5`       | `-6`  |
| `<<`  | 左移      | `5 << 2`   | `20`  |
| `>>`  | 右移（有符号） | `20 >> 2`  | `5`   |
| `>>>` | 右移（无符号） | `-1 >>> 1` | 很大的正数 |

### 2. 按位运算详解

```java
// 5 的二进制：0101
// 3 的二进制：0011

// 按位与 &：都为1才为1
System.out.println(5 & 3);  // 1 (0001)

// 按位或 |：有1则1
System.out.println(5 | 3);  // 7 (0111)

// 按位异或 ^：不同为1，相同为0
System.out.println(5 ^ 3);  // 6 (0110)

// 按位取反 ~：0变1，1变0
System.out.println(~5);  // -6
// 5 = 00000101
// ~5 = 11111010（补码表示 -6）
```

### 3. 移位运算详解

```java
// 左移 <<：相当于乘以 2^n
System.out.println(5 << 1);   // 10（5 * 2^1）
System.out.println(5 << 2);   // 20（5 * 2^2）
System.out.println(5 << 3);   // 40（5 * 2^3）

// 右移 >>：相当于除以 2^n（保留符号位）
System.out.println(20 >> 1);  // 10（20 / 2^1）
System.out.println(20 >> 2);  // 5（20 / 2^2）
System.out.println(-20 >> 2); // -5（保留负号）

// 无符号右移 >>>：高位补0
System.out.println(-1 >>> 1); // 2147483647（Integer.MAX_VALUE）
```

### 4. 位运算的应用

```java
// 1. 判断奇偶（比 % 效率高）
int n = 7;
if ((n & 1) == 0) {
    System.out.println("偶数");
} else {
    System.out.println("奇数");
}

// 2. 交换两个数（不用临时变量）
int a = 10, b = 20;
a = a ^ b;
b = a ^ b;
a = a ^ b;
System.out.println("a = " + a + ", b = " + b);  // a = 20, b = 10

// 3. 快速乘除 2 的幂
int x = 10;
System.out.println(x << 1);  // 20（x * 2）
System.out.println(x << 2);  // 40（x * 4）
System.out.println(x >> 1);  // 5（x / 2）

// 4. 获取某一位的值
int num = 0b10110;  // 二进制
int bit2 = (num >> 2) & 1;  // 获取第2位（从0开始）

// 5. 设置某一位为1
int num2 = 0b10110;
num2 = num2 | (1 << 0);  // 设置第0位为1

// 6. 清除某一位（设为0）
int num3 = 0b10110;
num3 = num3 & ~(1 << 1);  // 清除第1位
```

### 5. 位运算总结

| 运算    | 表达式                           | 说明      |
| ----- | ----------------------------- | ------- |
| 判断奇偶  | `n & 1`                       | 0为偶，1为奇 |
| 乘以2   | `n << 1`                      | 左移1位    |
| 除以2   | `n >> 1`                      | 右移1位    |
| 乘以2^k | `n << k`                      | 左移k位    |
| 除以2^k | `n >> k`                      | 右移k位    |
| 交换两数  | `a^=b; b^=a; a^=b`            | 不用临时变量  |
| 取绝对值  | `(n ^ (n >> 31)) - (n >> 31)` | 适用于int  |

## 九、三元运算符（条件运算符）

### 1. 基本语法

```java
数据类型 变量名 = 条件表达式 ? 值1 : 值2;
```

**规则：** 条件为 true，取值1；条件为 false，取值2

### 2. 使用示例

```java
public class TernaryDemo {
    public static void main(String[] args) {
        // 基本使用
        int a = 10;
        int b = 20;
        int max = (a > b) ? a : b;
        System.out.println("最大值：" + max);  // 20
        
        // 判断奇偶
        int num = 7;
        String result = (num % 2 == 0) ? "偶数" : "奇数";
        System.out.println(result);  // 奇数
        
        // 判断成绩
        int score = 75;
        String level = (score >= 60) ? "及格" : "不及格";
        System.out.println(level);  // 及格
    }
}
```

### 3. 三元运算符嵌套

```java
// 求三个数的最大值
int a = 10, b = 20, c = 15;
int max = (a > b) ? ((a > c) ? a : c) : ((b > c) ? b : c);
System.out.println("最大值：" + max);  // 20

// 成绩等级判断
int score = 85;
String grade = (score >= 90) ? "优秀" : 
               (score >= 80) ? "良好" : 
               (score >= 60) ? "及格" : "不及格";
System.out.println(grade);  // 良好
```

### 4. 三元运算符与 if-else 对比

```java
// 三元运算符
int max = (a > b) ? a : b;

// 等价的 if-else
int max;
if (a > b) {
    max = a;
} else {
    max = b;
}

// 三元运算符更简洁，适合简单判断
// if-else 更灵活，适合复杂逻辑
```

### 5. 注意事项

```java
// 1. 三元运算符必须有返回值
// (a > b) ? a : b;  // ❌ 错误，必须接收结果
int max = (a > b) ? a : b;  // ✅ 正确

// 2. 两个结果的类型要兼容
int result = (true) ? 10 : 20;       // ✅ 都是 int
double result = (true) ? 10 : 20.5;  // ✅ 自动转换
// String result = (true) ? 10 : "hello";  // ❌ 类型不兼容

// 3. 可以用于输出
System.out.println((a > b) ? "a大" : "b大");  // ✅ 正确

// 4. 三元运算符的类型推断
Object obj = (true) ? 1 : "hello";  // ✅ 结果是 Object
var v = (true) ? 1 : 2.0;  // v 是 double 类型
```

| 优先级 | 运算符                            | 说明         |

| 5   | `<<` `>>` `>>>`                | 移位         |

| 8   | `&`                            | 按位与        |
| 9   | `^`                            | 按位异或       |
| 10  | `\|`                           | 按位或        |
| 11  | `&&`                           | 逻辑与        |
| 12  | `\|\|`                         | 逻辑或        |
| 13  | `? :`                          | 三元运算符      |
| 14  | `=` `+=` `-=` `*=` ...         | 赋值运算符      |

### 2. 简化记忆

    优先级从高到低：
    1. () 括号最高
    2. ! ~ ++ -- 单目运算符
    3. * / % 乘除
    4. + - 加减
    5. << >> >>> 移位
    6. 比较运算符 > < >= <= == !=
    7. 位运算符 & ^ |
    8. 逻辑运算符 && ||
    9. 三元运算符 ? :
    10. 赋值运算符 = += -= ...

### 3. 口诀

    括号优先级最高
    单目高于双目
    算术高于移位
    移位高于关系
    关系高于位运算
    位运算高于逻辑
    逻辑高于三元
    三元高于赋值

### 4. 建议

```java
// ❌ 不推荐：依赖优先级，难以阅读

// ✅ 推荐：使用括号明确优先级
int result = ((a + (b * c)) > d) && e || f;

// 或者拆分成多行
int temp = a + b * c;
boolean compare = temp > d;
boolean result = compare && e || f;
```

### 4. next() 与 nextLine() 的区别

### 5. 混用问题及解决方案

```java
Scanner sc = new Scanner(System.in);

// 问题：nextInt() 后使用 nextLine()
System.out.print("输入年龄：");
int age = sc.nextInt();  // 输入 123 后按回车

System.out.print("输入姓名：");
String name = sc.nextLine();  // 会直接读取上次残留的回车，得到空字符串

System.out.println("姓名：" + name);  // 输出为空

// 解决方案1：多调用一次 nextLine() 消耗回车

int age = sc.nextInt();
sc.nextLine();  // 消耗回车

System.out.print("输入姓名：");
String name = sc.nextLine();

// 解决方案2：统一使用 nextLine()，再转换类型
System.out.print("输入年龄：");
int age = Integer.parseInt(sc.nextLine());

System.out.print("输入姓名：");
String name = sc.nextLine();
```

### 7. 异常处理

```java
import java.util.Scanner;
import java.util.InputMismatchException;

public class ScannerExceptionDemo {

        try {
            System.out.print("请输入分数：");
            double score = sc.nextDouble();
            System.out.println("分数：" + score);
        } catch (InputMismatchException e) {
            System.out.println("输入格式错误！");
        }

        int number = 0;
        boolean valid = false;
        while (!valid) {
            System.out.print("请输入一个整数：");
            if (sc.hasNextInt()) {
                number = sc.nextInt();
                valid = true;
            } else {
                System.out.println("输入错误，请重新输入！");
                sc.next();  // 消耗错误输入
            }
        }
        System.out.println("你输入的是：" + number);
        
        sc.close();
    }
}
```

```java
import java.util.Scanner;

public class StudentScore {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        // 输入学生信息
        System.out.print("请输入姓名：");
        String name = sc.next();
        
        System.out.print("请输入语文成绩：");

        // 判断等级
        String level = (avg >= 90) ? "优秀" :
                       (avg >= 80) ? "良好" :

        int maxScore = Math.max(chinese, Math.max(math, english));
        String maxSubject = (maxScore == chinese) ? "语文" :
                           (maxScore == math) ? "数学" : "英语";

        System.out.println("姓名：" + name);
        System.out.println("语文：" + chinese);
        System.out.println("数学：" + math);
        System.out.println("英语：" + english);
        System.out.println("总分：" + total);
        System.out.printf("平均分：%.2f%n", avg);
        System.out.println("等级：" + level);
        System.out.println("状态：" + passInfo);
        System.out.println("最高分科目：" + maxSubject + "（" + maxScore + "分）");

    }
```
